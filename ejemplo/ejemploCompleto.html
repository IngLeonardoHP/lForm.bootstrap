3. Ejemplo HTML y JS

Validate form validateInput, lengthMax, lengthMin, disregard y clearInput

Codigo html

<fieldset id="contenedor">
<div class="form-group">
    <input class="form-control" placeholder="Minimo 2 caracteres" type="text" data-lengthMin="2">
</div>
<div class="form-group">
    <input class="form-control" placeholder="Maximo 6 caracteres" type="text" data-lengthMax="6">
</div>
<div class="form-group">
    <input class="form-control onlyNumber" placeholder="Ignorar y solo numeros" type="text" data-disregard="true">
</div>
<button class="btn btn-default" id="enviar">Enviar</button>
<button class="btn btn-default" id="Limpiar">Limpiar</button>
</fieldset>

Codigo js

$("#enviar").click(function(){
  var btn=$(this);
  if($.validateInput("#contenedor")){
      btn.loading("Registrando",true,true);
      setTimeout(function(){
          btn.restore();
          $.clearInput(".panel-body");
      },2000);
  }
});
$("#Limpiar").click(function(){
  $.clearInput("#contenedor");
});

4. Ejemplo HTML y JS

Form a Ajax lFormAjaxMagic

Vaciar campo

Codigo html

<form method="post" action="php/url.php" id="lForm" enctype="multipart/form-data">
  <div class="form-group">
      <input class="form-control" placeholder="Nombre" name="inputNombre" type="text">
  </div>
  <div class="form-group">
      <input class="form-control" placeholder="Cedula" name="inputCedula" type="text">
  </div>
  <div class="form-group">
    <input type="file" name="imagen" id="exampleInputFile">
  </div>
  <button class="login-lightbox btn btn-primary lForm-submit" id="registrar">Registrar </button>
  <a href="#!" class="login-lightbox btn btn-default lForm-clear">Vaciar campo</a>
</form>

Codigo js

$(document).ready(function(){
  $.lFormAjaxMagic("#lForm",callback,true,true);
})

function callback(datos){
  console.log(datos);
  alert("Listo!");
}
Convierte un form de html en ajax con "$.lFormAjaxMagic("#div",callback,true,true)". Debe tenerse en cuenta que el form debe estar bien estructurado, es decir, declarado el "method", "action", "name" (para los input), button con la clase ".lForm-submit", y opcional button con la clase ".lForm-clear"
